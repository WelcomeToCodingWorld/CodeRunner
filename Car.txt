CREATE TABLE IF NOT EXISTS brand (id INT(4) AUTO_INCREMENT PRIMARY KEY,name VARCHAR(20) NOT NULL,level char(1) NOT NULL);

CREATE TABLE IF NOT EXISTS car (id INT(4) AUTO_INCREMENT PRIMARY KEY,brand_id INT(4) NOT NULL,sub_brand_id INT(4) NOT NULL,series_id INT(4) NOT NULL,type_id INT(4) NOT NULL,year char(4) NOT NULL,engin_displacement VARCHAR(3) NOT NULL,gearbox_type VARCHAR(10) NOT NULL);

CREATE TABLE IF NOT EXISTS sub_brand (id INT(4) AUTO_INCREMENT PRIMARY KEY,name VARCHAR(20) NOT NULL,brand_id INT(4) NOT NULL);

CREATE TABLE IF NOT EXISTS car_series (id INT(4) AUTO_INCREMENT PRIMARY KEY,name VARCHAR(20) NOT NULL,sub_brand_id INT(4) NOT NULL);

CREATE TABLE IF NOT EXISTS car_type (id INT(4) AUTO_INCREMENT PRIMARY KEY,name VARCHAR(20) NOT NULL,series_id INT(4) NOT NULL);
